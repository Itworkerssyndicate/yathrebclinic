🔒 قواعد Firestore (Lock Mode - ممنوع الوصول إلا للمصرح لهم) 
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // الدوال المساعدة
    function isAuthenticated() {
      return request.auth != null;
    }
    
    function isSuperAdmin() {
      return isAuthenticated() && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'super_admin';
    }
    
    function isAdmin() {
      return isAuthenticated() && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role in ['super_admin', 'admin'];
    }
    
    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }
    
    // 🔒 قواعد صارمة للمستخدمين
    match /users/{userId} {
      allow read: if isSuperAdmin() || isOwner(userId);
      allow write: if isSuperAdmin(); // فقط السوبر أدمن يعدل
      allow create: if isSuperAdmin(); // منع إنشاء حسابات عشوائية
      allow delete: if isSuperAdmin(); // فقط السوبر أدمن يحذف
    }
    
    // 🔒 قواعد الإعدادات - الأهمية القصوى
    match /settings/{document} {
      allow read: if true; // أي حد يقرأ (لظهور الموقع)
      allow write: if isSuperAdmin(); // فقط السوبر أدمن يعدل
      allow create: if isSuperAdmin();
      allow delete: if false; // ممنوع الحذف نهائياً
    }
    
    // 🔒 قواعد الخدمات الطبية
    match /clinicServices/{service} {
      allow read: if true;
      allow write: if isAdmin(); // المشرفين يعدلوا
      allow create: if isAdmin();
      allow delete: if isSuperAdmin(); // الحذف للسوبر أدمن فقط
    }
    
    // 🔒 قواعد تحاليل المعمل
    match /labTests/{test} {
      allow read: if true;
      allow write: if isAdmin();
      allow create: if isAdmin();
      allow delete: if isSuperAdmin();
    }
    
    // 🔒 قواعد العروض والقوافل
    match /offers/{offer} {
      allow read: if true;
      allow write: if isAdmin();
      allow delete: if isSuperAdmin();
    }
    
    // 🔒 قواعد الحضانات
    match /nurseries/{nursery} {
      allow read: if true;
      allow write: if isAdmin();
      allow delete: if isSuperAdmin();
    }
    
    // 🔒 قواعد صفحة الفيسبوك
    match /facebookPosts/{post} {
      allow read: if true;
      allow write: if isAdmin();
      allow delete: if isSuperAdmin();
    }
    
    // 🔒 قواعد سجل النشاطات
    match /activityLog/{log} {
      allow read: if isAdmin(); // المشرفين يشوفوا السجل
      allow write: if false; // ممنوع الكتابة المباشرة (بتتكتب تلقائياً)
      allow create: if isAuthenticated(); // فقط المستخدمين الموثوقين
    }
    
    // 🔒 قواعد الصلاحيات
    match /roles/{role} {
      allow read: if isSuperAdmin();
      allow write: if isSuperAdmin(); // فقط السوبر أدمن
    }
    
    // 🔒 رفض أي عمليات غير مصرح بها
    match /{document=**} {
      allow read, write: if false; // الباقي كله مرفوض
    }
  }
}
